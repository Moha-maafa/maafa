Backup script using rsync with logging (Bash)
#!/bin/bash

SOURCE="/home/student/"
DEST="/home/student/backup/"
LOGFILE="/home/student/backup.log"

mkdir -p "$DEST"

rsync -av --delete "$SOURCE" "$DEST" >> "$LOGFILE" 2>&1

echo "Backup completed on $(date)" >> "$LOGFILE"
