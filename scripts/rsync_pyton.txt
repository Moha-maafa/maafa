#!/usr/bin/env python3

import subprocess
import os

source = "/home/student/"
destination = "/backup/student_home/"

os.makedirs(destination, exist_ok=True)

subprocess.run([
    "rsync",
    "-av",
    "--delete",
    source,
    destination
])
